DECLARE
  v_min   NUMBER := 10;  -- Intervallum eleje
  v_max   NUMBER := 100; -- Intervallum vége
  v_ertek NUMBER := 78;  -- A vizsgált szám
BEGIN
  IF v_ertek BETWEEN v_min AND v_max THEN
    DBMS_OUTPUT.PUT_LINE(v_ertek || ' beleesik az intervallumba.');
  ELSE
    DBMS_OUTPUT.PUT_LINE(v_ertek || ' kívül esik az intervallumon.');
  END IF;
END;
